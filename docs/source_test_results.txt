class PublisherLayoutStorage {
  final AppDatabase db;

  PublisherLayoutStorage(this.db);

  Future<List<String>> loadOrder() async {
    final rows = await db.publisherLayoutDao.getLayout();
    return rows.map((e) => e.publisherId).toList();
  }

  Future<void> saveOrder(List<String> ids) async {
    await db.publisherLayoutDao.saveLayout(ids);
  }
}

publisher_layout_controller.dart
class PublisherLayoutController extends StateNotifier<List<String>> {
  final PublisherLayoutStorage storage;

  PublisherLayoutController(this.storage) : super([]);

  Future<void> load(List<String> defaultIds) async {
    final saved = await storage.loadOrder();
    state = saved.isEmpty ? defaultIds : saved;
  }

  void reorder(int from, int to) {
    final list = [...state];
    final item = list.removeAt(from);
    list.insert(to, item);
    state = list;
    storage.saveOrder(state);
  }
}

class DraggablePublisherGrid extends ConsumerWidget {
  final List<Publisher> publishers;

  const DraggablePublisherGrid({super.key, required this.publishers});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final order = ref.watch(publisherLayoutProvider);
    final controller = ref.read(publisherLayoutProvider.notifier);

    final ordered = order
        .map((id) => publishers.firstWhere((p) => p.id == id))
        .toList();

    return ReorderableWrap(
      spacing: 12,
      runSpacing: 12,
      needsLongPressDraggable: true,
      onReorder: controller.reorder,
      children: ordered.map((p) {
        return PublisherTile(publisher: p, key: ValueKey(p.id));
      }).toList(),
    );
  }
}
